   async sendAudio(jid: string, audio: Buffer, mimetype?: string): Promise<void> {
     if (!this.connected) {
       logger.warn({ jid }, 'Cannot send audio while disconnected');
       return;
     }
     try {
       await this.sock.sendMessage(jid, {
         audio,
         ptt: true,
         mimetype: mimetype || 'audio/ogg; codecs=opus',
       });
       logger.info({ jid, size: audio.length }, 'Voice note sent');
     } catch (err) {
       logger.error({ jid, err }, 'Failed to send voice note');
     }
   }
   ```

   Place it after the existing `sendMessage` method. Key detail: `ptt: true` makes it appear as an inline-playable voice note in WhatsApp, not a downloadable audio file. The Baileys API accepts `Buffer` as `WAMediaUpload`.

   Do NOT add queuing/retry logic for audio (unlike text messages). Audio is large and ephemeral — if the connection is down, log a warning and drop it. The caller can retry.

   export function routeOutboundAudio(
     channels: Channel[],
     jid: string,
     audio: Buffer,
     mimetype?: string,
   ): Promise<void> {
     const channel = channels.find((c) => c.ownsJid(jid) && c.isConnected());
     if (!channel) throw new Error(`No channel for JID: ${jid}`);
     if (!channel.sendAudio) throw new Error(`Channel ${channel.name} does not support audio`);
     return channel.sendAudio(jid, audio, mimetype);
   }
   ```
